#include<iostream>
#include<vector>
using namespace std;

static long long t = 0;

void mergeSort(vector<int>& a, vector<int>& b, int l, int r) {
    if (l >= r) return;
    int mid = (l + r) / 2;
    mergeSort(a, b, l, mid);
    mergeSort(a, b, mid + 1, r);
    int i = l, j = mid + 1, k = l;
    while (i <= mid && j <= r) {
        if (a[i] <= a[j]) b[k++] = a[i++];
        else {
            b[k++] = a[j++];
            t += (mid - i + 1); // ğŸ’¡ç»Ÿè®¡é€†åºå¯¹
        }
    }
    while (i <= mid) b[k++] = a[i++];
    while (j <= r) b[k++] = a[j++];
    for (int p = l; p <= r; p++) a[p] = b[p];
}

int main() {
    int n;
    cin >> n;
    vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    mergeSort(a, b, 0, n - 1);
    cout << t;
    return 0;
}
